insert into servlet_music_member values((select 'M' || to_char(sysdate,'yyyymmdd') || LPAD(nvl(max(substr(mno,11))+1,1),4,'0') from SERVLET_MUSIC_MEMBER where substr(mno,2,8) = to_char(sysdate,'yyyymmdd')),'email','password','name',sysdate,'gender','membershipPlan');

drop table SERVLET_MUSIC_MEMBER; create table SERVLET_MUSIC_MEMBER( mno varchar2(500) primary key, email varchar2(80) not null, password varchar2(20) not null, name varchar2(20) not null, birth varchar2(20) not null, gender varchar2(30), membershipPlan varchar2(20) not null ); select * from SERVLET_MUSIC_MEMBER;

drop table SERVLET_MUSIC_PRODUCT; create table SERVLET_MUSIC_PRODUCT( p_num number not null, p_title varchar2(500) primary key, p_date date, p_producer varchar2(100), p_text varchar2(500), p_img varchar2(500), p_email varchar2(500), p_seller varchar2(500) ); select * from servlet_music_product; create sequence seq_servlet_music_product;
